Principio 1 · Anti-Leakage

Todo fit/transform de preprocesamiento se aplica después del train/test split. El test set nunca contamina los parámetros aprendidos.

Principio 2 · Reproducibilidad

Semilla fija en todos los pasos estocásticos. Código versionable, parametrizable y ejecutable en cualquier entorno con el mismo resultado.

Principio 3 · Baseline First

Siempre comparar con el modelo más simple posible. La complejidad adicional solo se justifica si el beneficio es estadísticamente significativo.

# ── 5. DETECCIÓN DE OUTLIERS (MÉTODO IQR) ────────────────
def detect_outliers_iqr(df, multiplier=1.5):
    outlier_report = {}
    for col in df.select_dtypes('number').columns:
        Q1, Q3 = df[col].quantile([0.25, 0.75])
        IQR = Q3 - Q1
        n_out = ((df[col] < Q1 - multiplier*IQR) |
                  (df[col] > Q3 + multiplier*IQR)).sum()
        if n_out > 0:
            outlier_report[col] = n_out
    return outlier_report

    ('cat', categorical_transformer, categoric_features),

Evaluación Regresión — Suite Completa 📉 
from sklearn.metrics import (mean_squared_error, mean_absolute_error,
                               mean_absolute_percentage_error, r2_score)
import numpy as np

def regression_report(y_true, y_pred, label="SET"):
    rmse  = np.sqrt(mean_squared_error(y_true, y_pred))
    mae   = mean_absolute_error(y_true, y_pred)
    mape  = mean_absolute_percentage_error(y_true, y_pred) * 100
    r2    = r2_score(y_true, y_pred)
    pbias = 100 * (np.sum(y_pred - y_true) / np.sum(y_true))  # PBIAS

    # NSE (Nash-Sutcliffe Efficiency) — estándar hidrológico
    nse = 1 - (np.sum((y_true - y_pred)**2) /
               np.sum((y_true - np.mean(y_true))**2))

    # KGE (Kling-Gupta Efficiency)
    r     = np.corrcoef(y_true, y_pred)[0, 1]
    alpha = np.std(y_pred) / np.std(y_true)
    beta  = np.mean(y_pred) / np.mean(y_true)
    kge   = 1 - np.sqrt((r-1)**2 + (alpha-1)**2 + (beta-1)**2)

    print(f"\n{'═'*40}")
    print(f"  REPORTE DE EVALUACIÓN — {label}")
    print(f{'═'*40}")
    print(f"  RMSE  : {rmse:.4f}")
    print(f"  MAE   : {mae:.4f}")
    print(f"  MAPE  : {mape:.2f}%")
    print(f"  R²    : {r2:.4f}")
    print(f"  PBIAS : {pbias:.2f}%")
    print(f"  NSE   : {nse:.4f} {'✓' if nse > 0.65 else '✗'}")
    print(f"  KGE   : {kge:.4f} {'✓' if kge > 0.65 else '✗'}")
    return {"rmse":rmse, "nse":nse, "kge":kge, "r2":r2, "pbias":pbias}

# Uso
regression_report(y_train, model.predict(X_train), "TRAIN")
regression_report(y_test,  model.predict(X_test),  "TEST")

def classification_full_report(model, X, y, label="SET"):
    y_pred  = model.predict(X)
    y_proba = model.predict_proba(X)

    print(f"\n{'═'*50}")
    print(f"  REPORTE DE CLASIFICACIÓN — {label}")
    print(f{'═'*50}")

    # Reporte por clase
    print(classification_report(y, y_pred))

    # AUC-ROC (macro para multiclase)
    try:
        auc_roc = roc_auc_score(y, y_proba,
                                  multi_class='ovr', average='macro')
        print(f"  AUC-ROC (macro): {auc_roc:.4f}")
    except:
        pass

    # Verifica si modelo es mejor que baseline mayoritario
    from sklearn.dummy import DummyClassifier
    dummy = DummyClassifier(strategy='most_frequent').fit(X, y)
    dummy_acc = (dummy.predict(X) == y).mean()
    model_acc = (y_pred == y).mean()
    print(f"\n  Accuracy modelo:   {model_acc:.4f}")

# Diagnóstico de overfitting
def overfit_check(score_train, score_val, metric="metric", threshold=0.10):
    gap = score_train - score_val
    print(f"\n  [{metric}] Train: {score_train:.4f} | Val: {score_val:.4f} | Gap: {gap:.4f}")
    if gap > threshold:
        print("  ⚠ POSIBLE OVERFITTING → aumentar regularización o datos")
    elif gap < -threshold:
        print("  ⚠ POSIBLE UNDERFITTING → reducir regularización o modelo más complejo")
    else:
        print("  ✓ Generalización saludable")

from scipy.stats import ks_2samp, chi2_contingency
import warnings

def detect_data_drift(X_reference, X_new,
                        threshold=0.05):
    """
    Compara distribuciones de features entre datos de entrenamiento
    y datos nuevos usando test KS (numéricos) o chi² (categóricos).
    Alerta si p-value < threshold (drift detectado).
    """
    drift_report = {}
    for col in X_reference.columns:
        if X_reference[col].dtype in ['float64', 'int64']:
            stat, pval = ks_2samp(X_reference[col].dropna(),
                                    X_new[col].dropna())
            drift_report[col] = {
                "test": "KS", "p_value": round(pval, 4),
                "drift": pval < threshold
            }

    drifted = [k for k, v in drift_report.items() if v['drift']]
    if drifted:
        print(f"⚠ DRIFT DETECTADO en: {drifted}")
        print("  → Considera re-entrenamiento del modelo")
    else:
        print("✓ Sin drift significativo detectado")
    return drift_report

Señales de Alerta — Qué Verificar Siempre 🚨
  • Score train >> score test → overfitting. Regularizar, más datos, menos features.
  • Score train ≈ score test ≈ bajo → underfitting. Modelo más complejo, más features, eliminar regularización.
  • Score test > score baseline por margen insignificante → modelo no aporta valor real.
  • Alta varianza entre folds (std > 0.05) → modelo inestable. Revisar data quality o aumentar CV folds.
  • Accuracy ≈ proporción clase mayoritaria → el modelo no aprendió nada. Revisar desbalance.
  • PBIAS > 25% → sesgo sistemático. Revisar feature engineering y transformaciones.
  • Residuos con patrón sistemático → relación no capturada. Agregar features o modelo más complejo.
  • Feature importancia dispersa y uniforme → posible ruido o multicolinealidad. Revisitar EDA.
